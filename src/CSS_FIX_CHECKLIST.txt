# Проверка nginx
cat /etc/nginx/sites-available/ivo.conf | grep root
# Должно быть: root /var/www/ivo/build;

tail -30 /var/log/nginx/ivo_error.log

# Тест nginx
nginx -t

# Перезапуск nginx
systemctl restart nginx
